# Тестирование программы

LOAD 3 0       # Загружаем размер массива в регистр R3
INC 2 0 

CYCLE:
    LOAD 1 2       # Загружаем значение из текущей ячейки, на которую указывает R2
    ADD 0 1        # Складываем значение из R1 с текущей суммой в R0
    INC 2 0        # Увеличиваем указатель (R2) на 1
    CMP 2 3 # Сравниваем значения в 3м и 4м регистрах 
    JUMP_IF END #  Проверяем условие сравнения
    JUMP CYCLE      # Переход обратно к началу цикла

END: 
STORE 3 0       # Сохраняем итоговый результат в ячейку 3
HALT        # Завершаем выполнение программы
