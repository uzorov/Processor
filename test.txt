# Тестирование программы

LOAD 1 0        # Загружаем предел суммы (значение из ячейки 1) в регистр R1
INC 2 0         # Увеличиваем значение регистра R2 на 1 (начальный счётчик)
LOAD_SIZE 3 0   # Загружаем размер массива (значение из ячейки 3) в регистр R3

CYCLE:
    LOAD 1 2        # Загружаем значение из ячейки 1 в регистр R1
    ADD 0 1         # Складываем значение в R1 (из ячейки 1) с R0 (текущий результат)
    INC 2 0         # Увеличиваем счётчик в R2 на 1
    JUMP_IF 2 END   # Если счётчик (R2) достиг значения в R3, выходим из цикла
    JUMP CYCLE      # Если условие не выполнено, продолжаем цикл

END:
STORE 3 0       # Сохраняем итоговый результат в ячейку 3
HALT 0 0        # Завершаем выполнение программы
